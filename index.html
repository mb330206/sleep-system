🔧 Supabase 設定

前往 supabase.com 建立免費專案後， 在「Project Settings → API」取得以下資訊，並在資料庫中執行下方 SQL 建立資料表。

-- 在 Supabase SQL Editor 執行：
CREATE TABLE sleep_records (
  id bigint generated always as identity primary key,
  member_id text not null,
  date text not null,
  sleep_time text,
  wake_time text,
  quality integer,
  duration text,
  factors text[],
  note text,
  created_at timestamptz default now()
);

-- 啟用 Row Level Security（建議）
ALTER TABLE sleep_records ENABLE ROW LEVEL SECURITY;

-- 允許所有讀寫（如需驗證請自行調整）
CREATE POLICY "allow_all" ON sleep_records FOR ALL USING (true) WITH CHECK (true);
